32 * This file is included by <avr/io.h> whenever compiling for an Xmega
33 * device. It abstracts certain features common to the Xmega device
34 * families.
35 */

37#ifndef _AVR_XMEGA_H
38#define _AVR_XMEGA_H
39
40#ifdef __DOXYGEN__
41/**
42 \ingroup avr_io
43
44 Write value \c value to IO register \c reg that is protected through
45 the Xmega or AVRrc configuration change protection (CCP) mechanism. This
46 implements the timed sequence that is required for CCP.
47
48 This macro requires that the address of \c reg is a compile-time constant.
49 When that is not the case, the \c ccp_write_io() function can be used.
50
51 Example to modify the CPU clock:
52 \code
53 #include <avr/io.h>
54
55 _PROTECTED_WRITE (CLK_PSCTRL, CLK_PSADIV0_bm);
56 _PROTECTED_WRITE (CLK_CTRL, CLK_SCLKSEL0_bm);
57 \endcode */
58#define _PROTECTED_WRITE(reg, value)
59
60/**
61 \ingroup avr_io
62
63 Write value \c value to register \c reg that is protected through
64 the Xmega or ATtiny102/104 configuration change protection (CCP) key for self
65 programming (SPM). This implements the timed sequence that is
66 required for CCP.
67
68 This macro requires that the address of \c reg is a compile-time constant.
69 When that is not the case, the \c ccp_write_spm() function can be used.
70
71 Example to modify the CPU clock:
72 \code
73 #include <avr/io.h>
74
75 _PROTECTED_WRITE_SPM (NVMCTRL_CTRLA, NVMCTRL_CMD_PAGEERASEWRITE_gc);
76 \endcode */
77#define _PROTECTED_WRITE_SPM(reg, value)
78
79#else /* !__DOXYGEN__ */
80
81#ifdef __AVR_TINY__
82
83#define _PROTECTED_WRITE(reg, value) \
84 __asm__ __volatile__ ("out %i0, %1" "\n\t" \
85 "out %i2, %3" \
86 : \
87 : "n" (& CCP), \
88 "d" ((uint8_t) 0xd8), \
89 "n" (& (reg)), \
90 "r" ((uint8_t) (value)))
91
92#elif defined (__AVR_XMEGA__)
93
94#define _PROTECTED_WRITE(reg, value) \
95 __asm__ __volatile__ ("out %i0, %1" "\n\t" \
96 "sts %2, %3" \
97 : \
98 : "n" (& CCP), \
99 "d" ((uint8_t) CCP_IOREG_gc), \
100 "n" (& (reg)), \
101 "r" ((uint8_t) (value)))
102#endif /* AVR_TINY || Xmega */
