38/**
39 \defgroup compat_ina90 <compat/ina90.h>: Compatibility with IAR EWB 3.x
40
41 \code #include <compat/ina90.h> \endcode
42
43 This is an attempt to provide some compatibility with
44 header files that come with IAR C, to make porting applications
45 between different compilers easier. No 100% compatibility though.
46
47 \note For actual documentation, please see the IAR manual.
48 */

53#define _CLI() do { __asm__ __volatile__ ("cli"); } while (0)
54#define _SEI() do { __asm__ __volatile__ ("sei"); } while (0)
55#define _NOP() do { __asm__ __volatile__ ("nop"); } while (0)

58#define _OPC(op) do { __asm__ __volatile__ (".word %0" : : "n" (op)); } while (0)

65/* _EEGET, _EEPUT */
66#include <avr/eeprom.h>
67
68#define input(port) (port)
69#define output(port, val) do { (port) = (val); } while (0)
70
71#define __inp_blk__(port, addr, cnt, op) do { \
72 unsigned char __i = (cnt); \
73 unsigned char *__addr = (addr); \
74 while (__i) { \
75 *(__addr op) = input(port); \
76 __i--; \
77 } \
78 } while (0)
79
80#define input_block_inc(port, addr, cnt) __inp_blk__(port, addr, cnt, ++)
81#define input_block_dec(port, addr, cnt) __inp_blk__(port, addr, cnt, --)
82
83#define __out_blk__(port, addr, cnt, op) do { \
84 unsigned char __i = (cnt); \
85 const unsigned char *__addr = (addr); \
86 while (__i) { \
87 output(port, *(__addr op)); \
88 __i--; \
89 } \
90 } while (0)
91
92#define output_block_inc(port, addr, cnt) __out_blk__(port, addr, cnt, ++)
93#define output_block_dec(port, addr, cnt) __out_blk__(port, addr, cnt, --)
94
95#endif
