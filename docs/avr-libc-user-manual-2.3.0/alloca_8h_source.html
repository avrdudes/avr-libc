29#ifndef _ALLOCA_H
30#define _ALLOCA_H 1
31
32#include <stddef.h>

34/** \defgroup alloca <alloca.h>: Allocate space in the stack */
35
36/** \ingroup alloca
37 \brief Allocate \a __size bytes of space in the stack frame of the caller.
38
39 This temporary space is automatically freed when the function that
40 called alloca() returns to its caller. AVR-LibC defines the alloca() as
41 a macro, which is translated into the inlined \c __builtin_alloca()
42 function. The fact that the code is inlined, means that it is impossible
43 to take the address of this function, or to change its behaviour by
44 linking with a different library.
45
46 \return alloca() returns a pointer to the beginning of the allocated
47 space. If the allocation causes stack overflow, program behaviour is
48 undefined.
49
50 \warning Avoid use alloca() inside the list of arguments of a function
51 call.
52 */
53extern void *alloca (size_t __size);
54
55#define alloca(size) __builtin_alloca (size)
56
57#endif /* alloca.h */
