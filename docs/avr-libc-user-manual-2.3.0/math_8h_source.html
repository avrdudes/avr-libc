54/** \file */
55/** \defgroup avr_math <math.h>: Mathematics
56 \code #include <math.h> \endcode
57
58 This header file declares basic mathematics constants and
59 functions.
60
61 \par Notes:
62 - Math functions do not raise exceptions and do not change the
63 \c errno variable. Therefore the majority of them are declared
64 with \c const attribute, for better optimization by GCC.
65 - 64-bit floating-point arithmetic is only available in
66 <a href="https://gcc.gnu.org/gcc-10/changes.html#avr">GCC v10</a>
67 and up.
68 The size of the \c double and \c long \c double type can be selected
69 at compile-time with options like <tt>-mdouble=64</tt> and
70 <tt>-mlong-double=32</tt>. Whether such options are available,
71 and their default values,
72 depend on how the compiler has been configured.
73 - The implementation of 64-bit floating-point arithmetic has some
74 shortcomings and limitations, see the
75 <a href="https://gcc.gnu.org/wiki/avr-gcc#Libf7">avr-gcc Wiki</a>
76 for details.
77 - In order to access the <tt>float</tt> functions,
78 in avr-gcc v4.6 and older it is usually
79 also required to link with \c -lm. In avr-gcc v4.7 and up, \c -lm
80 is added automatically to all linker invocations.
81
82 See also sone benchmarks for \ref bench_libm "IEEE single"
83 and \ref bench_libf7 "IEEE double".
84*/

479/** The isfinitef() function returns a nonzero value if \a __x is finite:
480 not plus or minus infinity, and not NaN. */
481__ATTR_CONST__ static __ATTR_ALWAYS_INLINE__ int isfinitef (float __x)
482{
483 unsigned char __exp;
484 __asm__ (
485 "mov %0, %C1" "\n\t"
486 "lsl %0" "\n\t"
487 "mov %0, %D1" "\n\t"
488 "rol %0"
489 : "=&r" (__exp)
490 : "r" (__x) );
491 return __exp != 0xff;
492}

520/** The copysignf() function returns \a __x but with the sign of \a __y.
521 They work even if \a __x or \a __y are NaN or zero. */
522__ATTR_CONST__ static __ATTR_ALWAYS_INLINE__ float copysignf (float __x, float __y)
523{
524 __asm__ (
525 "bst %D2, 7" "\n\t"
526 "bld %D0, 7"
527 : "=r" (__x)
528 : "0" (__x), "r" (__y));
529 return __x;
530}
