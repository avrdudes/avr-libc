47/** \file */
48/** \defgroup util_delay <util/delay.h>: Convenience functions for busy-wait delay loops
49 \code
50 #define F_CPU 1000000UL // 1 MHz
51 //#define F_CPU 14.7456e6
52 #include <util/delay.h>
53 \endcode
54
55 \note As an alternative method, it is possible to pass the
56 F_CPU macro down to the compiler from the Makefile.
57 Obviously, in that case, no \c \#define statement should be
58 used.
59
60 The functions in this header file are wrappers around the basic
61 busy-wait functions from <tt><util/delay_basic.h></tt>. They are meant as
62 convenience functions where actual time values can be specified
63 rather than a number of cycles to wait for. The idea behind is
64 that compile-time constant expressions will be eliminated by
65 compiler optimization so floating-point expressions can be used
66 to calculate the number of delay cycles needed based on the CPU
67 frequency passed by the macro F_CPU.
68
69 \note In order for these functions to work as intended, compiler
70 optimizations <em>must</em> be enabled, and the delay time
71 <em>must</em> be an expression that is a known constant at
72 compile-time. If these requirements are not met, the resulting
73 delay will be much longer (and basically unpredictable), and
74 applications that otherwise do not use floating-point calculations
75 will experience severe code bloat by the floating-point library
76 routines linked into the application.
77
78 The functions available allow the specification of microsecond, and
79 millisecond delays directly, using the application-supplied macro
80 F_CPU as the CPU clock frequency (in Hertz).
81*/
82

89 \brief CPU frequency in Hz
90
91 The macro F_CPU specifies the CPU frequency to be considered by
92 the delay macros. This macro is normally supplied by the
93 environment (e.g. from within a project header, or the project's
94 Makefile). The value 1 MHz here is only provided as a "vanilla"
95 fallback if no such user-provided definition could be found.
96
97 In terms of the delay functions, the CPU frequency can be given as
98 a floating-point constant (e.g. 3.6864e6 for 3.6864 MHz).
99 However, the macros in <util/setbaud.h> require it to be an
100 integer value.
101 */

109/**
110 \ingroup util_delay
111
112 Perform a delay of \c __ms milliseconds, using _delay_loop_2().
113
114 The macro #F_CPU is supposed to be defined to a
115 constant defining the CPU clock frequency (in Hertz).
116
117 The maximal possible delay is 262.14 ms / F_CPU in MHz.
118
119 When the user request delay which exceed the maximum possible one,
120 _delay_ms() provides a decreased resolution functionality. In this
121 mode _delay_ms() will work with a resolution of 1/10&nbsp;ms, providing
122 delays up to 6.5535 seconds (independent from CPU frequency). The
123 user will not be informed about decreased resolution.
124
125 If the avr-gcc toolchain has \c __builtin_avr_delay_cycles()
126 support, maximal possible delay is 4294967.295 ms/ F_CPU in MHz. For
127 values greater than the maximal possible delay, overflow may result in
128 no delay i.e., 0&nbsp;ms.
129
130 Conversion of \c __ms into clock cycles may not always result in
131 an integral value. By default, the clock cycles are rounded up to the next
132 integer. This ensures that the user gets at least \c __ms
133 microseconds of delay.
134
135 Alternatively, by defining the macro \c __DELAY_ROUND_DOWN__, or
136 \c __DELAY_ROUND_CLOSEST__, before including this header file, the
137 algorithm can be made to round down, or round to closest integer,
138 respectively.
139
140 \note The implementation of _delay_ms() based on
141 \c __builtin_avr_delay_cycles() is not backward compatible with older
142 implementations. In order to get functionality backward compatible
143 with previous versions, the macro \c __DELAY_BACKWARD_COMPATIBLE__
144 must be defined before including this header file.
145 */
146static __ATTR_ALWAYS_INLINE__ void _delay_ms(double __ms);
147
148void
149_delay_ms(double __ms)
150{
151 double __tmp ;
152#if __HAS_DELAY_CYCLES && defined(__OPTIMIZE__) \
153 && !defined(__DELAY_BACKWARD_COMPATIBLE__)
154 uint32_t __ticks_dc;
155 __tmp = ((F_CPU) / 1e3) * __ms;
156
157 #if defined(__DELAY_ROUND_DOWN__)
158 __ticks_dc = (uint32_t)__builtin_fabs(__tmp);
159
160 #elif defined(__DELAY_ROUND_CLOSEST__)
161 __ticks_dc = (uint32_t)(__builtin_fabs(__tmp)+0.5);
162
163 #else
164 //round up by default
165 __ticks_dc = (uint32_t)(__builtin_ceil(__builtin_fabs(__tmp)));
166 #endif
167
168 __builtin_avr_delay_cycles(__ticks_dc);
169
170#else
171 uint16_t __ticks;
172 __tmp = ((F_CPU) / 4e3) * __ms;
173 if (__tmp < 1.0)
174 __ticks = 1;
175 else if (__tmp > 65535)
176 {
177 // __ticks = requested delay in 1/10 ms
178 __ticks = (uint16_t) (__ms * 10.0);
179 while(__ticks)
180 {
181 // wait 1/10 ms
182 _delay_loop_2(((F_CPU) / 4e3) / 10);
183 __ticks --;
184 }
185 return;
186 }
187 else
188 __ticks = (uint16_t)__tmp;
189 _delay_loop_2(__ticks);
190#endif
191}

231_delay_us(double __us)
232{
233 double __tmp ;
234#if __HAS_DELAY_CYCLES && defined(__OPTIMIZE__) \
235 && !defined(__DELAY_BACKWARD_COMPATIBLE__)
236 uint32_t __ticks_dc;
237 __tmp = ((F_CPU) / 1e6) * __us;
238
239 #if defined(__DELAY_ROUND_DOWN__)
240 __ticks_dc = (uint32_t)__builtin_fabs(__tmp);
241
242 #elif defined(__DELAY_ROUND_CLOSEST__)
243 __ticks_dc = (uint32_t)(__builtin_fabs(__tmp)+0.5);
244
245 #else
246 //round up by default
247 __ticks_dc = (uint32_t)(__builtin_ceil(__builtin_fabs(__tmp)));
248 #endif
249
250 __builtin_avr_delay_cycles(__ticks_dc);
251
252#else
253 uint8_t __ticks;
254 double __tmp2 ;
255 __tmp = ((F_CPU) / 3e6) * __us;
256 __tmp2 = ((F_CPU) / 4e6) * __us;
257 if (__tmp < 1.0)
258 __ticks = 1;
259 else if (__tmp2 > 65535)
260 {
261 _delay_ms(__us / 1000.0);
262 return;
263 }
264 else if (__tmp > 255)
265 {
266 uint16_t __ticks=(uint16_t)__tmp2;
267 _delay_loop_2(__ticks);
268 return;
269 }
270 else
271 __ticks = (uint8_t)__tmp;
272 _delay_loop_1(__ticks);
273#endif
274}
